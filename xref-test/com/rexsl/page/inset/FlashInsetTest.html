159                                     "Rexsl-Flash=U0VWRVJFOi0xOmV4IOKCrA;"
160                                 )
161                             ),
162                             new CustomMatcher<String>("valid forward cookie") {
163                                 @Override
164                                 public boolean matches(final Object obj) {
165                                     final FlashInset.Flash flash =
166                                         FlashInsetTest.flash(obj);
167                                     return flash.level().equals(Level.SEVERE)
168                                         && flash.message().endsWith(" \u20ac")

218     /**
219      * Convert object to a flash cookie.
220      * @param object The object (a text)
221      * @return Flash cookie
222      */
223     private static FlashInset.Flash flash(final Object object) {
224         final HttpCookie cookie =
225             HttpCookie.parse(object.toString()).get(0);
226         return new FlashInset.Flash(
227             new NewCookie("/xx", cookie.getValue())
228         );
229     }
