85       * @return Request

93       * BaseRequest can fetch HTTP request and process HTTP response.
94       * @throws Exception If something goes wrong inside
95       */
96      @Test
97      public void sendsHttpRequestAndProcessesHttpResponse() throws Exception {
98          final ContainerMocker container = new ContainerMocker()
99              .expectRequestUri(Matchers.containsString("helloall"))
100             .expectMethod(Matchers.equalTo(Request.GET))
101             .returnBody("hello!!".getBytes(CharEncoding.UTF_8))
102             .mock();
103         this.request(container.home())
104             .uri().path("/helloall").back()
105             .method(Request.GET)
106             .fetch().as(RestResponse.class)
107             .assertBody(Matchers.containsString("!!"))
108             .assertBody(Matchers.containsString("hello!"))
109             .assertStatus(HttpURLConnection.HTTP_OK);
110     }
