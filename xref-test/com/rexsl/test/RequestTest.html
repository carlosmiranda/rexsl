107     @Test
108     public void sendsHttpRequestAndProcessesHttpResponse() throws Exception {
109         final MkContainer container = new MkGrizzlyContainer().next(
110             new MkAnswer.Simple("\u20ac! hello!")
111         ).start();
112         this.request(container.home())
113             .uri().path("/helloall").back()
114             .method(Request.GET)
115             .fetch().as(RestResponse.class)
116             .assertBody(Matchers.containsString("\u20ac!"))
117             .assertBody(Matchers.containsString("hello!"))
118             .assertStatus(HttpURLConnection.HTTP_OK);
119         final MkQuery query = container.take();
120         MatcherAssert.assertThat(
121             query.uri(),
122             Matchers.hasToString(Matchers.containsString("helloall"))
123         );
124         MatcherAssert.assertThat(
125             query.method(),
126             Matchers.equalTo(Request.GET)
127         );
128         container.stop();
129     }
