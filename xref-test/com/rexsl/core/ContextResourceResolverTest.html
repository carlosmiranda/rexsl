67      /**
68       * ContextResourceResolver can resolve resource by HREF.
69       * @throws Exception If something goes wrong
70       */
71      @Test
72      public void resolvesResourceByHref() throws Exception {
73          final String href = "/test.xml?123";
74          final String ref = "/test.xml";
75          final ServletContext ctx = new ServletContextMocker()
76              .withResource(ref, "<r>\u0443</r>")
77              .mock();
78          final URIResolver resolver = new ContextResourceResolver(ctx);
79          final Source src = resolver.resolve(href, null);
80          MatcherAssert.assertThat(src.getSystemId(), Matchers.equalTo(ref));
81          MatcherAssert.assertThat(src, XhtmlMatchers.hasXPath("/r[.='\u0443']"));
82      }

137             .when(conn).getInputStream();

197     /**
198      * ContextResourceResolver can resolve a file.
199      * @throws Exception If something goes wrong
200      */
201     @Test
202     public void resolvesUrlWithFile() throws Exception {
203         final File file = this.temp.newFile("file.xml");
204         FileUtils.writeStringToFile(file, "<data/>");
205         final String href = file.toURI().toString();
206         final ServletContext ctx = new ServletContextMocker().mock();
207         final URIResolver resolver = new ContextResourceResolver(ctx);
208         final Source src = resolver.resolve(href, null);
209         MatcherAssert.assertThat(src.getSystemId(), Matchers.equalTo(href));
210         MatcherAssert.assertThat(src, XhtmlMatchers.hasXPath("/data"));
211     }
