115     /**
116      * PageAnalyzer can make decide to transform.
117      * @throws Exception If something goes wrong
118      */
119     @Test
120     public void decidesToTransform() throws Exception {
121         final HttpServletRequestMocker mocker = new HttpServletRequestMocker();
122         if (this.agent != null) {
123             mocker.withHeader(HttpHeaders.USER_AGENT, this.agent);
124         }
125         if (this.accept != null) {
126             mocker.withHeader(HttpHeaders.ACCEPT, this.accept);
127         }
128         MatcherAssert.assertThat(
129             new PageAnalyzer(this.page, mocker.mock()).needsTransformation(),
130             Matchers.describedAs(
131                 Logger.format(
132                     "'%s' with '%s' by '%s'",
133                     this.page,
134                     this.accept,
135                     this.agent
136                 ),
137                 Matchers.equalTo(this.decision)
138             )
139         );
140     }
