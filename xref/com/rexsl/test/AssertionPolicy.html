32  import java.lang.annotation.Documented;
33  import java.lang.annotation.ElementType;
34  import java.lang.annotation.Retention;
35  import java.lang.annotation.RetentionPolicy;
36  import java.lang.annotation.Target;
37  import javax.validation.constraints.NotNull;

39  /**
40   * Policy of assertion, used by {@link TestResponse}.
41   *
42   * <p>Assertion policy is used in {@link TestResponse} in order to validate
43   * REST response before further processing. Consider this example:
44   *
45   * <pre>
46   * String data = RestTester.start(new URI("http://example.com"))
47   *   .assertThat(retryOnFailure)
48   *   .assertXPath("//data")
49   *   .xpath("//data/text()")
50   *   .get(0);
51   * </pre>
52   *
53   * <p>This code will 1) retrieve the document from the provided
54   * {@link java.net.URI}, 2) make sure that it has an XML node addressed by
55   * {@code "//data"} XPath, 3) retrieve all nodes by that XPath, and
56   * 4) use the first element in the found collection of texts. But, before
57   * doing all this a {@code retryOnFailure} assertion policy will be used:
58   *
59   * <pre> AssertionPolicy retryOnFailure = new AssertionPolicy() {
60   *   private boolean valid;
61   *   &#64;Override
62   *   public void assertThat(TestResponse response) {
63   *     if (response.getStatus() != HttpURLConnection.HTTP_OK) {
64   *       throw new AssertionError("invalid HTTP status, will try again");
65   *     }
66   *     this.valid = true;
67   *   }
68   *   &#64;Override
69   *   public boolean isRetryNeeded(int attempt) {
70   *     return !this.valid;
71   *   }
72   * }</pre>
73   *
74   * <p>{@link AssertionPolicy#assertThat(TestResponse)} will be called right
75   * after the document is retrieved from the server. Actually, it will be called
76   * even earlier, even before the HTTP request is made. In the example above
77   * the request will be made inside {@code response.getStatus()}. If it will
78   * fail for some reason (host not found, for example) - an exception will be
79   * thrown and {@code this.valid} will stay in {@code FALSE} status. Thus, it
80   * is a bullet-proof design, where you can control everything, including
81   * network errors.
82   *
83   * <p>Implementation of this interface need NOT be thread-safe.
84   *
85   * @author Yegor Bugayenko (yegor@tpc2.com)
86   * @version $Id$
87   * @since 0.3.4
88   */
89  public interface AssertionPolicy {
90  
91      /**
92       * Annotates a policy that should not produce a full exception trace
93       * in log if it fails.
94       */
95      @Documented
96      @Retention(RetentionPolicy.RUNTIME)
97      @Target(ElementType.TYPE)
98      public @interface Quiet {
99      }
100 
101     /**
102      * Make an assertion and return nothing or throw {@link AssertionError}
103      * if some problem is found.
104      *
105      * <p>Validity information about the response should be collected in
106      * this method and stored in the object's variable. Later you should use
107      * it in {@link #isRetryNeeded(int)} in order to inform
108      * the caller of whether it
109      * should retry the request or not.
110      *
111      * <p>If the method doesn't throw {@link AssertionError} it means that
112      * everything went fine and {@link #isRetryNeeded(int)} will
113      * never be called.
114      *
115      * @param response The response to assert
116      */
117     void assertThat(@NotNull TestResponse response);
118 
119     /**
120      * Do we need to re-fetch the page and try again?
121      *
122      * <p>This method is called by
123      * {@link TestResponse#assertThat(AssertionPolicy)} before the next
124      * attempt, in order to check whether it is required. {@link TestResponse}
125      * keeps track of attempts (so you don't have to do it here) and will stop
126      * making them after {@link TestResponse#MAX_ATTEMPTS}. In most cases
127      * you don't need to store this attempt number locally.
128      * If {@code retryAgain()} returns {@code TRUE} {@link TestResponse}
129      * will try again, otherwise it will throw an exception.
130      *
131      * @param attempt Number of attempt we're trying to make (will not ask for
132      *  attempt #0, but will start from #1)
133      * @return Yes, we should try again (if {@code TRUE})
134      */
135     boolean isRetryNeeded(int attempt);
136 
137 }
