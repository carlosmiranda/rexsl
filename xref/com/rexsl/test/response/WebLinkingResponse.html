50  /**
51   * Web Linking response.
52   *
53   * <p>This response decorator is able to understand and parse {@code Link}
54   * HTTP header according to
55   * <a href="http://tools.ietf.org/html/rfc5988">RFC 5988 "Web Linking"</a>,
56   * for example:
57   *
58   * <pre> String name = new JdkRequest("http://my.example.com")
59   *   .fetch()
60   *   .as(WebLinkingResponse.class)
61   *   .follow("next")
62   *   .fetch();</pre>
63   *
64   * <p>The class is immutable and thread-safe.
65   *
66   * @author Yegor Bugayenko (yegor@tpc2.com)
67   * @version $Id$
68   * @since 0.9
69   * @see <a href="http://tools.ietf.org/html/rfc5988">RFC 5988 "Web Linking"</a>
70   */
71  @Immutable
72  @EqualsAndHashCode(callSuper = true)
73  @SuppressWarnings("PMD.TooManyMethods")
74  public final class WebLinkingResponse extends AbstractResponse {
75  
76      /**
77       * ImmutableHeader name.
78       */
79      private static final String HEADER = "Link";
80  
81      /**
82       * Param name.
83       */
84      private static final String REL = "rel";
85  
86      /**
87       * Public ctor.
88       * @param resp Response
89       */
90      public WebLinkingResponse(
91          @NotNull(message = "response can't be NULL") final Response resp) {
92          super(resp);
93      }
94  
95      /**
96       * Follow link by REL.
97       * @param rel Relation name
98       * @return The same object
99       * @throws IOException If fails
100      */
101     @NotNull(message = "response is never NULL")
102     public Request follow(@NotNull(message = "rel can't be NULL")
103         final String rel) throws IOException {
104         final WebLinkingResponse.Link link = this.links().get(rel);
105         if (link == null) {
106             throw new IOException(
107                 String.format(
108                     "Link with rel=\"%s\" doesn't exist, use #hasLink()",
109                     rel
110                 )
111             );
112         }
113         return new RestResponse(this).jump(link.uri());
114     }
115 
116     /**
117      * Get all links provided.
118      * @return List of all links found
119      * @throws IOException If fails
120      */
121     @NotNull(message = "list of links is never NULL")
122     @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
123     public Map<String, WebLinkingResponse.Link> links() throws IOException {
124         final ConcurrentMap<String, WebLinkingResponse.Link> links =
125             new ConcurrentHashMap<String, WebLinkingResponse.Link>();
126         final Collection<String> headers =
127             this.headers().get(WebLinkingResponse.HEADER);
128         if (headers != null) {
129             for (final String header : headers) {
130                 for (final String part : header.split(",")) {
131                     final WebLinkingResponse.Link link =
132                         new WebLinkingResponse.SimpleLink(part.trim());
133                     final String rel = link.get(WebLinkingResponse.REL);
134                     if (rel != null) {
135                         links.put(rel, link);
136                     }
137                 }
138             }
139         }
140         return links;
141     }

155     /**
156      * Implementation of a link.
157      */
158     @Immutable
159     @EqualsAndHashCode
160     private static final class SimpleLink implements WebLinkingResponse.Link {
161         /**
162          * Pattern to match link value.
163          */
164         private static final Pattern PTN = Pattern.compile(
165             "<([^>]+)>\\s*;(.*)"
166         );
167         /**
168          * URI encapsulated.
169          */
170         private final transient String addr;
171         /**
172          * Map of link params.
173          */
174         private final transient ArrayMap<String, String> params;
175         /**
176          * Public ctor (parser).
177          * @param text Text to parse
178          * @throws IOException If fails
179          */
180         SimpleLink(final String text) throws IOException {
181             final ConcurrentMap<String, String> args =
182                 new ConcurrentHashMap<String, String>();
183             final Matcher matcher = WebLinkingResponse.SimpleLink.PTN
184                 .matcher(text);
185             if (!matcher.matches()) {
186                 throw new IOException(
187                     String.format(
188                         "Link header value doesn't comply to RFC-5988: \"%s\"",
189                         text
190                     )
191                 );
192             }
193             this.addr = matcher.group(1);
194             for (final String pair
195                 : matcher.group(2).trim().split("\\s*;\\s*")) {
196                 final String[] parts = pair.split("=");
197                 args.put(
198                     parts[0].trim().toLowerCase(Locale.ENGLISH),
199                     StringUtils.strip(parts[1].trim(), "\"")
200                 );
201             }
202             this.params = new ArrayMap<String, String>(args);
203         }
204         @Override
205         public URI uri() {
206             return URI.create(this.addr);
207         }
208         @Override
209         public int size() {
210             return this.params.size();
211         }
212         @Override
213         public boolean isEmpty() {
214             return this.params.isEmpty();
215         }
216         @Override
217         public boolean containsKey(final Object key) {
218             return this.params.containsKey(key);
219         }
220         @Override
221         public boolean containsValue(final Object value) {
222             return this.params.containsValue(value);
223         }
224         @Override
225         public String get(final Object key) {
226             return this.params.get(key);
227         }
228         @Override
229         public String put(final String key, final String value) {
230             throw new UnsupportedOperationException("#put()");
231         }
232         @Override
233         public String remove(final Object key) {
234             throw new UnsupportedOperationException("#remove()");
235         }
236         @Override
237         public void putAll(final Map<? extends String, ? extends String> map) {
238             throw new UnsupportedOperationException("#putAll()");
239         }
240         @Override
241         public void clear() {
242             throw new UnsupportedOperationException("#clear()");
243         }
244         @Override
245         public Set<String> keySet() {
246             return this.params.keySet();
247         }
248         @Override
249         public Collection<String> values() {
250             return this.params.values();
251         }
252         @Override
253         public Set<Map.Entry<String, String>> entrySet() {
254             return this.params.entrySet();
255         }
256     }
