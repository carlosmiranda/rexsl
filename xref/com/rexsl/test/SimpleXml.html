82      /**
83       * Public ctor, from XML as a text.
84       *
85       * <p>The object is created with a default implementation of
86       * {@link javax.xml.namespace.NamespaceContext}, which already defines a
87       * number of namespaces, for convenience, including:
88       *
89       * <pre> xhtml: http://www.w3.org/1999/xhtml
90       * xs: http://www.w3.org/2001/XMLSchema
91       * xsi: http://www.w3.org/2001/XMLSchema-instance
92       * xsl: http://www.w3.org/1999/XSL/Transform
93       * svg: http://www.w3.org/2000/svg</pre>
94       *
95       * <p>In future versions we will add more namespaces (submit a ticket if
96       * you need more of them defined here).
97       *
98       * <p>An {@link IllegalArgumentException} is thrown if the parameter
99       * passed is not in XML format.
100      *
101      * @param text Body
102      */
103     public SimpleXml(@NotNull final String text) {
104         this(
105             new DomParser(text).document().getDocumentElement(),
106             new XPathContext()
107         );
108     }

227     /**
228      * {@inheritDoc}
229      */
230     @Override
231     @NotNull
232     public List<String> xpath(@NotNull final String query) {
233         final NodeList nodes = this.nodelist(query);
234         final List<String> items = new ArrayList<String>(nodes.getLength());
235         for (int idx = 0; idx < nodes.getLength(); ++idx) {
236             MatcherAssert.assertThat(
237                 "Only /text() nodes or attributes are retrievable with xpath()",
238                 nodes.item(idx).getNodeType(),
239                 Matchers.<Short>either(Matchers.equalTo(Node.TEXT_NODE))
240                     .or(Matchers.equalTo(Node.ATTRIBUTE_NODE))
241                     .or(Matchers.equalTo(Node.CDATA_SECTION_NODE))
242             );
243             items.add(nodes.item(idx).getNodeValue());
244         }
245         return new ListWrapper<String>(items, this.dom, query);
246     }

258     /**

260      */
261     @Override
262     @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
263     @NotNull
264     public List<XmlDocument> nodes(@NotNull final String query) {
265         final NodeList nodes = this.nodelist(query);
266         final List<XmlDocument> items =
267             new ArrayList<XmlDocument>(nodes.getLength());
268         for (int idx = 0; idx < nodes.getLength(); ++idx) {
269             items.add(new SimpleXml(nodes.item(idx), this.context));
270         }
271         return new ListWrapper<XmlDocument>(items, this.dom, query);
272     }

283     /**
284      * Retrieve and return a nodelist for XPath query.
285      *
286      * <p>An {@link IllegalArgumentException} is thrown if the parameter
287      * passed is not a valid XPath expression.
288      *
289      * @param query XPath query
290      * @return List of DOM nodes
291      */
292     private NodeList nodelist(final String query) {
293         NodeList nodes;
294         try {
295             final XPath xpath = XPathFactory.newInstance().newXPath();
296             xpath.setNamespaceContext(this.context);
297             nodes = (NodeList) xpath.evaluate(
298                 query,
299                 this.dom,
300                 XPathConstants.NODESET
301             );
302         } catch (javax.xml.xpath.XPathExpressionException ex) {
303             throw new IllegalArgumentException(
304                 String.format("invalid XPath query '%s'", query),
305                 ex
306             );
307         }
308         return nodes;
309     }

311     /**
312      * Transform source to DOM node.
313      * @param source The source
314      * @return The node
315      */
316     private static Node transform(final Source source) {
317         final DOMResult result = new DOMResult();
318         try {
319             TransformerFactory.newInstance()
320                 .newTransformer()
321                 .transform(source, result);
322         } catch (javax.xml.transform.TransformerConfigurationException ex) {
323             throw new IllegalStateException(ex);
324         } catch (javax.xml.transform.TransformerException ex) {
325             throw new IllegalStateException(ex);
326         }
327         return result.getNode();
328     }
