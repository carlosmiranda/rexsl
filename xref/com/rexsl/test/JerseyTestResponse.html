185     @Override
186     @NotNull
187     @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
188     public Cookie cookie(@NotNull final String name) {
189         final MultivaluedMap<String, String> headers = this.getHeaders();
190         MatcherAssert.assertThat(
191             "cookies should be set in HTTP header",
192             headers.containsKey(HttpHeaders.SET_COOKIE)
193         );
194         final String header = StringUtils.join(
195             headers.get(HttpHeaders.SET_COOKIE), ", "
196         );
197         Cookie cookie = null;
198         for (HttpCookie candidate : HttpCookie.parse(header)) {
199             if (candidate.getName().equals(name)) {
200                 cookie = new Cookie(
201                     candidate.getName(),
202                     candidate.getValue(),
203                     candidate.getPath(),
204                     candidate.getDomain(),
205                     candidate.getVersion()
206                 );
207                 break;
208             }
209         }
210         MatcherAssert.assertThat(
211             Logger.format(
212                 "cookie '%s' not found in Set-Cookie header: '%s'",
213                 name, header
214             ),
215             cookie,
216             Matchers.notNullValue()
217         );
218         assert cookie != null;
219         return cookie;
220     }

247     @Override
248     @NotNull
249     public TestResponse assertThat(@NotNull final AssertionPolicy assertion) {
250         synchronized (this.fetcher) {
251             int attempt = 0;
252             while (true) {
253                 try {
254                     assertion.assertThat(this);
255                     break;
256                 } catch (AssertionError ex) {
257                     ++attempt;
258                     if (!assertion.isRetryNeeded(attempt)) {
259                         throw ex;
260                     }
261                     if (attempt >= JerseyTestResponse.MAX_ATTEMPTS) {
262                         this.fail(
263                             Logger.format("failed after %d attempt(s)", attempt)
264                         );
265                     }
266                     if (assertion.getClass()
267                         .getAnnotation(AssertionPolicy.Quiet.class) == null) {
268                         Logger.warn(
269                             this,
270                             // @checkstyle LineLength (1 line)
271                             "#assertThat(%[type]s): attempt #%d failed, re-trying: %[exception]s",
272                             assertion, attempt, ex
273                         );
274                     } else {
275                         Logger.warn(
276                             this,
277                             "#assertThat(%[type]s): attempt #%d: %s",
278                             assertion, attempt, ex.getMessage()
279                         );
280                     }
281                     this.fetcher.reset();
282                 }
283             }
284             return this;
285         }
286     }

354     /**
355      * Follow the URI provided.
356      * @param uri The URI to follow
357      * @return New client
358      */
359     @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
360     private TestClient follow(final URI uri) {
361         final TestClient client = RestTester.start(uri);
362         for (NewCookie cookie : this.fetcher.fetch().getCookies()) {
363             if (!cookie.getValue().isEmpty()) {
364                 client.header(
365                     HttpHeaders.COOKIE,
366                     new Cookie(cookie.getName(), cookie.getValue()).toString()
367                 );
368             }
369         }
370         return client;
371     }
