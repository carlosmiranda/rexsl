110     /**

112      * @param wre Wire
113      * @param uri The resource to work with
114      * @param headers Headers
115      * @param method HTTP method
116      * @param body HTTP request body

119     BaseRequest(final Wire wre, final String uri,

165     @Override
166     @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
167     public Response fetch() throws IOException {
168         final long start = System.currentTimeMillis();
169         final Collection<Map.Entry<String, String>> headers =
170             new LinkedList<Map.Entry<String, String>>();
171         boolean agent = false;
172         for (final Map.Entry<String, String> header : this.hdrs) {
173             headers.add(new Header(header.getKey(), header.getValue()));
174             if (header.getKey().equals(HttpHeaders.USER_AGENT)) {
175                 agent = true;
176             }
177         }
178         if (!agent) {
179             headers.add(new Header(HttpHeaders.USER_AGENT, BaseRequest.AGENT));
180         }
181         final String info = URI.create(this.home).getUserInfo();
182         if (info != null) {
183             final String[] parts = info.split(":", 2);
184             try {
185                 headers.add(
186                     new Header(
187                         HttpHeaders.AUTHORIZATION,
188                         Logger.format(
189                             "Basic %s",
190                             Base64.encodeBase64String(
191                                 Logger.format(
192                                     "%s:%s",
193                                     URLDecoder.decode(
194                                         parts[0], CharEncoding.UTF_8
195                                     ),
196                                     URLDecoder.decode(
197                                         parts[1], CharEncoding.UTF_8
198                                     )
199                                 ).getBytes()
200                             )
201                         )
202                     )
203                 );
204             } catch (UnsupportedEncodingException ex) {
205                 throw new IllegalStateException(ex);
206             }
207         }
208         final Response response = this.wire.send(
209             this, this.home, this.mtd, headers, this.content
210         );
211         Logger.info(
212             this,
213             "#fetch(%s %s): completed in %[ms]s [%d %s]: %s",
214             this.mtd,
215             URI.create(this.home).getPath(),
216             System.currentTimeMillis() - start,
217             response.status(),
218             response.reason(),
219             this.home
220         );
221         return response;
222     }

224     @Override
225     public String toString() {
226         final StringBuilder text = new StringBuilder("HTTP/1.1 ")
227             .append(this.mtd).append(' ')
228             .append(URI.create(this.home).getPath())
229             .append('\n');
230         for (final Map.Entry<String, String> header : this.hdrs) {
231             text.append(
232                 Logger.format(
233                     "%s: %s\n",
234                     header.getKey(),
235                     header.getValue()
236                 )
237             );
238         }
239         text.append('\n');
240         if (this.content.isEmpty()) {
241             text.append("<<empty request body>>");
242         } else {
243             text.append(this.content);
244         }
245         return text.toString();
246     }

248     /**
249      * Apache URI.
250      */
251     @Immutable
252     @EqualsAndHashCode(of = "address")
253     private static final class ApacheURI implements RequestURI {
254         /**
255          * URI encapsulated.
256          */
257         private final transient String address;
258         /**
259          * Apache request encapsulated.
260          */
261         private final transient BaseRequest owner;
262         /**
263          * Public ctor.
264          * @param req Request
265          * @param uri The URI to start with
266          */
267         public ApacheURI(final BaseRequest req, final String uri) {
268             this.owner = req;
269             this.address = uri;
270         }
271         @Override
272         public String toString() {
273             return this.address;
274         }
275         @Override
276         public Request back() {
277             return new BaseRequest(
278                 this.owner.wire,
279                 this.address,
280                 this.owner.hdrs,
281                 this.owner.mtd,
282                 this.owner.content
283             );
284         }
285         @Override
286         public URI get() {
287             return URI.create(this.owner.home);
288         }
289         @Override
290         public RequestURI set(@NotNull(message = "URI can't be NULL")
291             final URI uri) {
292             return new BaseRequest.ApacheURI(this.owner, uri.toString());
293         }
294         @Override
295         public RequestURI queryParam(
296             @NotNull(message = "param name can't be NULL") final String name,
297             @NotNull(message = "value can't be NULL") final Object value) {
298             return new BaseRequest.ApacheURI(
299                 this.owner,
300                 UriBuilder.fromUri(this.address)
301                     .queryParam(name, "{value}")
302                     .build(value).toString()
303             );
304         }
305         @Override
306         public RequestURI path(
307             @NotNull(message = "path can't be NULL") final String segment) {
308             return new BaseRequest.ApacheURI(
309                 this.owner,
310                 UriBuilder.fromUri(this.address)
311                     .path(segment)
312                     .build().toString()
313             );
314         }
315     }

342             return this.text;
343         }
344         @Override
345         public Request back() {
346             return new BaseRequest(
347                 this.owner.wire,
348                 this.owner.home,
349                 this.owner.hdrs,
350                 this.owner.mtd,
351                 this.text
352             );
353         }
354         @Override
355         public String get() {
356             return this.text;
357         }
358         @Override
359         public RequestBody set(@NotNull(message = "content can't be NULL")
360             final String txt) {
361             return new BaseRequest.ApacheBody(this.owner, txt);
362         }
363         @Override
364         public RequestBody formParam(
365             @NotNull(message = "name can't be NULL") final String name,
366             @NotNull(message = "value can't be NULL") final Object value) {
367             try {
368                 return new BaseRequest.ApacheBody(
369                     this.owner,
370                     new StringBuilder(this.text)
371                         .append(name)
372                         .append('=')
373                         .append(
374                             URLEncoder.encode(
375                                 value.toString(), CharEncoding.UTF_8
376                             )
377                         )
378                         .append('&')
379                         .toString()
380                 );
381             } catch (UnsupportedEncodingException ex) {
382                 throw new IllegalStateException(ex);
383             }
384         }
