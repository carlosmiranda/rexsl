43  /**
44   * Wrapper of {@link List}.
45   *
46   * <p>This wrapper is our internal implementation of a {@link List}. The only
47   * purpose of this wrapper is to throw our own custom exception when the client
48   * is trying to access an element that is absent in the list. Such a custom
49   * exception ({@link ListWrapper.NodeNotFoundException})
50   * includes detailed information about
51   * the original document. Thus, such an incorrect list-access operation will
52   * lead to an exception that contains all the details inside (not just a simple
53   * error message). For example:
54   *
55   * <pre>String name = RestTester.start(uri)
56   *   .get("load XML document")
57   *   .xpath("/document/name/text()")
58   *   .get(0)</pre>
59   *
60   * <p>This snippet is trying to get a text value from XML element, and there
61   * is no guarantee that such an element exists in the document. In order to give
62   * a detailed report to the user of the problem (including a full content of
63   * the XML document) we're returning {@link ListWrapper} from
64   * {@link JerseyTestResponse#xpath(String)}. The only method that we implement
65   * for this purpose is {@link #get(int)}.
66   *
67   * <p>{@link ListWrapper} is an unmodifiable list, that's why
68   * the majority of inherited method are not implemented and
69   * thow runtime exceptions if being called.
70   *
71   * <p>The class is immutable and thread-safe.
72   *
73   * @author Yegor Bugayenko (yegor@tpc2.com)
74   * @version $Id$
75   * @since 0.3.7
76   */

173     @Override
174     @NotNull
175     public T get(final int index) {
176         if (index >= this.size()) {
177             throw new ListWrapper.NodeNotFoundException(
178                 String.format(
179                     "Index (%d) is out of bounds (size=%d)",
180                     index,
181                     this.size()
182                 ),
183                 this.dom,
184                 this.xpath
185             );
186         }
187         return this.original.get(index);

289     /**
290      * {@inheritDoc}
291      *
292      * <p>The method throws {@link ListWrapper.NodeNotFoundException}
293      * when either
294      * {@code start} or {@code end} is bigger than the size of the list. In all
295      * other cases of illegal method call (start is less than zero, end is
296      * less than zero, or start is bigger than end) a standard
297      * {@link IndexOutOfBoundException} is thrown (by the encapsulated
298      * implementation of {@Link List}).
299      */
300     @Override
301     @NotNull
302     public List<T> subList(final int start, final int end) {
303         if (start >= this.size()) {
304             throw new ListWrapper.NodeNotFoundException(
305                 String.format(
306                     "Start of subList (%d) is out of bounds (size=%d)",
307                     start,
308                     this.size()
309                 ),
310                 this.dom,
311                 this.xpath
312             );
313         }
314         if (end >= this.size()) {
315             throw new ListWrapper.NodeNotFoundException(
316                 String.format(
317                     "End of subList (%d) is out of bounds (size=%d)",
318                     end,
319                     this.size()
320                 ),
321                 this.dom,
322                 this.xpath
323             );
324         }
325         return this.original.subList(start, end);
326     }

344     /**
345      * Node not found in XmlDocument.
346      */
347     private static final class NodeNotFoundException
348         extends IndexOutOfBoundsException {
349         /**
350          * Serialization marker.
351          */
352         private static final long serialVersionUID = 0x7526FA78EEDAC470L;
353         /**
354          * Public ctor.
355          * @param message Error message
356          * @param node The XML with error
357          * @param xpath The address
358          */
359         protected NodeNotFoundException(@NotNull final String message,
360             @NotNull final Node node, @NotNull final String xpath) {
361             super(
362                 Logger.format(
363                     "XPath '%s' not found in '%s': %s",
364                     StringEscapeUtils.escapeJava(xpath),
365                     // @checkstyle LineLength (1 line)
366                     StringEscapeUtils.escapeJava(new DomPrinter(node).toString()),
367                     message
368             )
369             );
370         }
371     }
