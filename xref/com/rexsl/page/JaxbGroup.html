54  /**
55   * JAXB group of elements.
56   *
57   * <p>A convenient utility class that enables on-fly creation of JAXB annotated
58   * collections of elements, for example:
59   *
60   * <pre> &#64;XmlRootElement
61   * public class Page {
62   *   &#64;XmlElement
63   *   public Object getEmployees() {
64   *     Collection&lt;Employee&gt; employees = this.retrieve();
65   *     return JaxbGroup.build(employees, "employee");
66   *   }
67   * }</pre>
68   *
69   * <p>It's even more convenient in combination with {@link JaxbBundle}, for
70   * example:
71   *
72   * <pre> &#64;XmlRootElement
73   * public class Page {
74   *   &#64;XmlElement
75   *   public Object getEmployee() {
76   *     return new JaxbBundle("employee")
77   *       .attr("age", "45")
78   *       .add("depts", JaxbGroup.build(this.depts(), "dept"))
79   *       .up()
80   *       .element();
81   *   }
82   * }</pre>
83   *
84   * <p>The class is immutable and thread-safe.
85   *
86   * @author Yegor Bugayenko (yegor@tpc2.com)
87   * @version $Id$
88   * @since 0.3.7
89   */
90  @XmlRootElement
91  @XmlAccessorType(XmlAccessType.NONE)
92  @ToString
93  public final class JaxbGroup {

96       * Classes already created before.
97       */
98      private static final ConcurrentMap<String, Class<?>> READY =
99          new ConcurrentHashMap<String, Class<?>>(0);

102      * Collection of elements.
103      */
104     @NotNull
105     private final transient Collection<?> group;

108      * Public ctor, for JAXB (always throws a runtime exception).

110     public JaxbGroup() {
111         throw new IllegalStateException(

117     /**
118      * Protected ctor, for on-fly instantiating (you're not supposed to use it).
119      * @param grp Group of elements
120      */
121     protected JaxbGroup(final Collection<?> grp) {
122         this.group = grp;
123     }

125     /**
126      * Creates a new JAXB-annotated collection of elements.
127      * @param grp Group of elements (JAXB-annotated)
128      * @param name Name of parent XML element
129      * @return JAXB-annotated object, just created
130      */
131     @NotNull
132     public static Object build(@NotNull final Collection<?> grp,
133         @NotNull final String name) {
134         synchronized (JaxbGroup.READY) {
135             final String mnemo = JaxbGroup.mnemo(grp.isEmpty(), name);
136             if (!JaxbGroup.READY.containsKey(mnemo)) {
137                 JaxbGroup.READY.put(
138                     mnemo,
139                     JaxbGroup.construct(JaxbGroup.types(grp), name)
140                 );
141             }
142             try {
143                 return JaxbGroup.READY.get(mnemo)
144                     .getDeclaredConstructor(Collection.class)
145                     .newInstance(grp);
146             } catch (NoSuchMethodException ex) {
147                 throw new IllegalStateException(ex);
148             } catch (InstantiationException ex) {
149                 throw new IllegalStateException(ex);
150             } catch (IllegalAccessException ex) {
151                 throw new IllegalStateException(ex);
152             } catch (java.lang.reflect.InvocationTargetException ex) {
153                 throw new IllegalStateException(ex);
154             }
155         }
156     }

169     /**
170      * Create NEW class name.
171      * @param empty Is it an empty group?
172      * @param name Name of root element
173      * @return The name
174      */
175     private static String mnemo(final boolean empty, final String name) {
176         return String.format(
177             "%s$%s$%s",
178             JaxbGroup.class.getName(),
179             name,
180             // @checkstyle AvoidInlineConditionals (1 line)
181             empty ? "empty" : "full"
182         );
183     }

185     /**
186      * Construct new class.
187      * @param types Types used in the collection
188      * @param name Name of root element
189      * @return Class just created
190      */
191     private static Class<?> construct(final Collection<Class<?>> types,
192         final String name) {
193         final ClassPool pool = ClassPool.getDefault();
194         try {
195             final CtClass ctc = pool.getAndRename(
196                 JaxbGroup.class.getName(),
197                 JaxbGroup.mnemo(types.isEmpty(), name)
198             );
199             final ClassFile file = ctc.getClassFile();
200             final AnnotationsAttribute attribute =
201                 (AnnotationsAttribute) file.getAttribute(
202                     AnnotationsAttribute.visibleTag
203                 );
204             attribute.addAnnotation(JaxbGroup.xmlRootElement(file, name));
205             if (!types.isEmpty()) {
206                 attribute.addAnnotation(JaxbGroup.xmlSeeAlso(file, types));
207             }
208             final Class<?> cls = ctc.toClass();
209             ctc.defrost();
210             Logger.debug(
211                 JaxbGroup.class,
212                 "#construct('%s'): class %s constructed",
213                 name,
214                 cls.getName()
215             );
216             return cls;
217         } catch (javassist.NotFoundException ex) {
218             throw new IllegalStateException(ex);
219         } catch (javassist.CannotCompileException ex) {
220             throw new IllegalStateException(ex);
221         }
222     }

224     /**
225      * Find all types used in the collection.
226      * @param group The collection
227      * @return List of types used there
228      */
229     private static Collection<Class<?>> types(final Collection<?> group) {
230         final Collection<Class<?>> types = new HashSet<Class<?>>();
231         for (Object element : group) {
232             types.add(element.getClass());
233         }
234         return types;
235     }

237     /**
238      * Create new <tt>XmlRootElement</tt> annotation.
239      * @param file Javassist file to work with
240      * @param name Name of root element
241      * @return The annotation
242      */
243     private static Annotation xmlRootElement(final ClassFile file,
244         final String name) {
245         final AnnotationsAttribute attribute = AnnotationsAttribute.class.cast(
246             file.getAttribute(AnnotationsAttribute.visibleTag)
247         );
248         final Annotation annotation = attribute.getAnnotation(
249             XmlRootElement.class.getName()
250         );
251         annotation.addMemberValue(
252             "name",
253             new StringMemberValue(name, file.getConstPool())
254         );
255         Logger.debug(
256             JaxbGroup.class,
257             "#xmlRootElement(.., '%s'): annotation created",
258             name
259         );
260         return annotation;
261     }

263     /**
264      * Create new <tt>XmlSeeAlso</tt> annotation.
265      * @param file Javassist file to work with
266      * @param types The class to refer to
267      * @return The annotation
268      */
269     @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
270     private static Annotation xmlSeeAlso(final ClassFile file,
271         final Collection<Class<?>> types) {
272         final Annotation annotation = new Annotation(
273             XmlSeeAlso.class.getName(),
274             file.getConstPool()
275         );
276         final ArrayMemberValue member = new ArrayMemberValue(
277             file.getConstPool()
278         );
279         final ClassMemberValue[] values = new ClassMemberValue[types.size()];
280         int pos = 0;
281         for (Class<?> type : types) {
282             values[pos] = new ClassMemberValue(
283                 type.getName(),
284                 file.getConstPool()
285             );
286             pos += 1;
287         }
288         member.setValue(values);
289         annotation.addMemberValue("value", member);
290         Logger.debug(
291             JaxbGroup.class,
292             "#xmlSeeAlso(.., %d classes): annotation created",
293             types.size()
294         );
295         return annotation;
296     }
