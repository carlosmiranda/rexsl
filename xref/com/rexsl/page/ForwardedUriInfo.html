46  /**
47   * Custom implementation of {@link UriInfo} that is aware of
48   * {@code X-Forwarded-For} HTTP header.
49   *
50   * <p>The class is mutable and NOT thread-safe.
51   *
52   * @author Yegor Bugayenko (yegor@tpc2.com)
53   * @version $Id$
54   * @see <a href="http://tools.ietf.org/html/draft-ietf-appsawg-http-forwarded-10">IETF Forwarded HTTP Extension</a>
55   */
56  @ToString
57  final class ForwardedUriInfo implements UriInfo {
58  
59      /**
60       * Original {@link UriInfo}.
61       */
62      private final transient UriInfo info;
63  
64      /**
65       * Http headers, injected by JAX-RS implementation.
66       */
67      private final transient AtomicReference<HttpHeaders> headers;
68  
69      /**
70       * TRUE if HTTP headers already where analyzed.
71       */
72      private transient boolean analyzed;
73  
74      /**
75       * New host, or empty string if not required, or NULL if not yet sure.
76       */
77      private transient String host;
78  
79      /**
80       * Scheme to set, or NULL if not necessary.
81       */
82      private transient String scheme;
83  
84      /**
85       * Public ctor.
86       * @param inf The original UriInfo
87       * @param hdrs HTTP headers
88       */
89      public ForwardedUriInfo(final UriInfo inf,
90          final AtomicReference<HttpHeaders> hdrs) {
91          if (inf == null) {
92              throw new IllegalStateException(
93                  "UriInfo is incorrectly injected into BaseResource"
94              );
95          }
96          this.info = inf;
97          this.headers = hdrs;
98      }

255     /**
256      * Forward this builder to the right host/port (if necessary).
257      * @param builder The builder to forward
258      * @return The same builder
259      */
260     private UriBuilder forward(final UriBuilder builder) {
261         if (!this.analyzed) {
262             if (this.headers.get() == null) {
263                 throw new IllegalStateException(
264                     "HttpHeaders is not injected into BaseResource"
265                 );
266             }
267             for (Map.Entry<String, List<String>> header
268                 : this.headers.get().getRequestHeaders().entrySet()) {
269                 for (String value : header.getValue()) {
270                     this.consume(header.getKey(), value);
271                 }
272             }
273             Logger.debug(
274                 this,
275                 "#forward(..): analyzed, host=%s, scheme=%s",
276                 this.host,
277                 this.scheme
278             );
279             this.analyzed = true;
280         }
281         if (this.host != null) {
282             builder.host(this.host);
283         }
284         if (this.scheme != null) {
285             builder.scheme(this.scheme);
286         }
287         return builder;
288     }

290     /**
291      * Interpret HTTP header and save host/scheme pair into this object.
292      * @param name HTTP header name
293      * @param value HTTP header value
294      * @see <a href="http://tools.ietf.org/html/draft-ietf-appsawg-http-forwarded-10">IETF Forwarded HTTP Extension</a>
295      */
296     private void consume(final String name, final String value) {
297         if (this.host == null
298             && "x-forwarded-host".equals(name.toLowerCase(Locale.ENGLISH))) {
299             this.host = value;
300         } else if (this.scheme == null
301             && "x-forwarded-proto".equals(name.toLowerCase(Locale.ENGLISH))) {
302             this.scheme = value;
303         } else if ("forwarded".equals(name.toLowerCase(Locale.ENGLISH))) {
304             this.forwarded(value);
305         }
306     }
307 
308     /**
309      * Consume specifically "Forwarded" header.
310      * @param value HTTP header value
311      */
312     private void forwarded(final String value) {
313         for (String sector : value.split("\\s*,\\s*")) {
314             for (String opt : sector.split("\\s*;\\s*")) {
315                 final String[] parts = opt.split("=", 2);
316                 if (this.host == null && "host".equals(parts[0])) {
317                     this.host = parts[1];
318                 }
319                 if (this.scheme == null && "proto".equals(parts[0])) {
320                     this.scheme = parts[1];
321                 }
322             }
323         }
324     }
