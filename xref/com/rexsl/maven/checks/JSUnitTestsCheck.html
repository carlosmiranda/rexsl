54  /**
55   * Checks JS Unit tests.
56   *
57   * <p>Since this class is NOT public its documentation is not available online.
58   * All details of this check should be explained in the JavaDoc of
59   * {@link DefaultChecksProvider}.
60   *
61   * <p>The class is immutable and thread-safe.
62   *
63   * @author Evgeniy Nyavro (e.nyavro@gmail.com)
64   * @version $Id$
65   */
66  @ToString
67  @EqualsAndHashCode
68  final class JSUnitTestsCheck implements Check {
69  
70      /**
71       * Directory with JS test files.
72       */
73      private static final String JS_TESTS_DIR = "src/test/rexsl/js";
74  
75      /**
76       * Wrapper of rhino unit script.
77       */
78      private static final String RHINO_WRAPPER = "rhinoUnitWrapper.js";
79  
80      /**
81       * Rhino unit script.
82       */
83      private static final String RHINO_UNIT = "rhinoUnitUtil.js";
84  
85      @Override
86      @Loggable(Loggable.DEBUG)
87      public void setScope(@NotNull final String scope) {
88          // nothing to scope here
89      }
90  
91      @Override
92      @Loggable(Loggable.DEBUG)
93      public boolean validate(@NotNull final Environment env) {
94          final File dir = new File(env.basedir(), JSUnitTestsCheck.JS_TESTS_DIR);
95          boolean success = true;
96          if (dir.exists()) {
97              final Collection<File> files = new FileFinder(dir, "js").random();
98              for (File file : files) {
99                  final String name =
100                     FilenameUtils.removeExtension(file.getName());
101                 LoggingManager.enter(name);
102                 success &= this.one(env.basedir(), file);
103                 LoggingManager.leave();
104                 if (!success) {
105                     Logger.warn(
106                         this,
107                         "Validation Failed!"
108                     );
109                 }
110             }
111         } else {
112             Logger.info(
113                 this,
114                 "%s directory is absent, no JS unit tests",
115                 JSUnitTestsCheck.JS_TESTS_DIR
116             );
117         }
118         return success;
119     }
120 
121     /**
122      * Check one script.
123      * @param base Folder to check
124      * @param file JS Unit test file to check
125      * @return Is js unit test succeeded?
126      */
127     private boolean one(final File base, final File file) {
128         boolean valid;
129         final Reader reader = new BufferedReader(
130             new InputStreamReader(
131                 this.getClass().getResourceAsStream(
132                     JSUnitTestsCheck.RHINO_WRAPPER
133                 )
134             )
135         );
136         final ScriptEngineManager factory = new ScriptEngineManager();
137         final ScriptEngine engine = factory.getEngineByName("JavaScript");
138         try {
139             engine.put(
140                 "rhinoUnitUtil",
141                 IOUtil.toString(
142                     this.getClass().getResourceAsStream(
143                         JSUnitTestsCheck.RHINO_UNIT
144                     )
145                 )
146             );
147             engine.put("base", base.getAbsolutePath());
148             try {
149                 engine.eval(reader);
150             } finally {
151                 reader.close();
152             }
153             final Invocable invocable = (Invocable) engine;
154             final Object res = invocable.invokeFunction(
155                 "runTest",
156                 file.getAbsolutePath()
157             );
158             for (String msg : res.toString().split("\n")) {
159                 Logger.error(this, msg);
160             }
161             valid = res.toString().isEmpty();
162         } catch (NoSuchMethodException ex) {
163             this.logException(ex);
164             valid = false;
165         } catch (ScriptException ex) {
166             this.logException(ex);
167             valid = false;
168         } catch (IOException ex) {
169             this.logException(ex);
170             valid = false;
171         }
172         return valid;
173     }
174 
175     /**
176      * Logs exception to the logger.
177      * @param exc Exception to log.
178      */
179     private void logException(final Exception exc) {
180         Logger.error(
181             this,
182             "Failed:\n%[exception]s",
183             exc
184         );
185     }
