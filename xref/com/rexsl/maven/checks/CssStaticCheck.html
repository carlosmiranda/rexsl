44  /**
45   * Validates CSS files against style sheet rules.
46   *
47   * <p>The class is immutable and thread-safe.
48   *
49   * <p>Since this class is NOT public its documentation is not available online.
50   * All details of this check should be explained in the JavaDoc of
51   * {@link DefaultChecksProvider}.
52   *
53   * <p>{@code CssLint.class} file is created by Rhino JavaScript-to-Java
54   * compiler during {@code process-sources} Maven phase (see {@code pom.xml}
55   * file). Here we're just executing this class in a standalone process, in
56   * order to capture its output into string.
57   *
58   * @author Dmitry Bashkin (dmitry.bashkin@rexsl.com)
59   * @author Yegor Bugayenko (yegor@tpc2.com)
60   * @version $Id$
61   * @see <a href="http://www.mozilla.org/rhino/jsc.html">Rhino JavaScript to Java compiler</a>
62   * @see <a href="https://github.com/stubbornella/csslint/wiki/Command-line-interface">CSSLint Command Line Interface</a>
63   */
64  @ToString
65  @EqualsAndHashCode
66  final class CssStaticCheck implements Check {
67  
68      @Override
69      @Loggable(Loggable.DEBUG)
70      public void setScope(@NotNull final String scope) {
71          // nothing to scope here
72      }
73  
74      @Override
75      @Loggable(Loggable.DEBUG)
76      public boolean validate(@NotNull final Environment env) throws IOException {
77          final File csslint = new File(
78              this.getClass().getResource("/CssLint.class").getFile()
79          );
80          final ProcessBuilder builder = new ProcessBuilder(
81              "java",
82              "-classpath",
83              StringUtils.join(
84                  env.classpath(false),
85                  System.getProperty("path.separator")
86              ),
87              "CssLint",
88              env.basedir().getPath(),
89              "--format=compact"
90          );
91          builder.directory(csslint.getParentFile());
92          String report;
93          try {
94              final Process process = builder.start();
95              if (process.waitFor() != 0) {
96                  throw new IllegalStateException(
97                      IOUtils.toString(process.getErrorStream())
98                  );
99              }
100             report = IOUtils.toString(process.getInputStream());
101         } catch (InterruptedException ex) {
102             Thread.currentThread().interrupt();
103             throw new IllegalStateException(ex);
104         }
105         return this.isValid(report);
106     }
107 
108     /**
109      * Validate the report from CSSLint.
110      *
111      * <p>CSSLint report is a plain text document, where every line is a
112      * message about one defect. In this method we split this plain text
113      * document into lines and fetch them one by one to the log. If there are
114      * no lines in the document we return {@code true}, which means that
115      * there are no errors and the CSS document is valid.
116      *
117      * @param report The report (one problem per line)
118      * @return True if it's valid (no errors)
119      */
120     private boolean isValid(final String report) {
121         final String[] lines = report.split("\n");
122         for (final String line : lines) {
123             Logger.warn(this, "%s", line.trim());
124         }
125         return lines.length == 0;
126     }
