30  package com.rexsl.core;
31  
32  import com.jcabi.aspects.Loggable;
33  import com.jcabi.log.Logger;
34  import javax.servlet.http.HttpServletRequest;
35  import javax.ws.rs.core.HttpHeaders;
36  import javax.ws.rs.core.MediaType;
37  import lombok.EqualsAndHashCode;
38  import lombok.ToString;
39  
40  /**
41   * Makes a decision whether page should be transformed to HTML or returned
42   * to the user as untouched XML (or anything else).
43   *
44   * @author Yegor Bugayenko (yegor@tpc2.com)
45   * @version $Id$
46   */
47  @ToString
48  @EqualsAndHashCode(of = { "page", "request" })
49  @Loggable(Loggable.DEBUG)
50  final class PageAnalyzer {
51  
52      /**
53       * The page.
54       */
55      private final transient String page;
56  
57      /**
58       * The request.
59       */
60      private final transient HttpServletRequest request;
61  
62      /**
63       * Public ctor.
64       * @param text The text of response
65       * @param rqst The request
66       */
67      protected PageAnalyzer(final String text, final HttpServletRequest rqst) {
68          this.page = text;
69          this.request = rqst;
70      }
71  
72      /**
73       * Do we need filtering?
74       * @return Do we need to transform to XHTML?
75       */
76      public boolean needsTransformation() {
77          final UserAgent agent = new UserAgent(
78              this.request.getHeader(HttpHeaders.USER_AGENT)
79          );
80          final TypesMatcher accept = new TypesMatcher(
81              this.request.getHeader(HttpHeaders.ACCEPT)
82          );
83          // @checkstyle BooleanExpressionComplexity (1 line)
84          final boolean needs = this.notEmpty() && this.containsXml()
85              && !this.xmlDemanded(accept)
86              && !this.xslAccepted(agent, accept);
87          Logger.debug(
88              this,
89              // @checkstyle LineLength (1 line)
90              "#needsTransformation('%s': '%[text]s'): User-Agent='%s', Accept='%s', %B",
91              this.request.getRequestURI(),
92              this.page,
93              agent,
94              accept,
95              needs
96          );
97          return needs;
98      }
99  
100     /**
101      * Page require transformation, since it's not an empty page?
102      * @return TRUE if the page requires transformation
103      */
104     private boolean notEmpty() {
105         return !this.page.isEmpty();
106     }
107 
108     /**
109      * Page require transformation, since it contains XML and XSL stylesheet?
110      * @return TRUE if the page requires transformation
111      */
112     private boolean containsXml() {
113         return this.page.startsWith("<?xml ")
114             && this.page.contains("<?xml-stylesheet ");
115     }
116 
117     /**
118      * Page requires transformation, since XML media type is not required?
119      * @param types Media types
120      * @return TRUE if the page requires transformation
121      */
122     private boolean xmlDemanded(final TypesMatcher types) {
123         return types.explicit(MediaType.APPLICATION_XML);
124     }
125 
126     /**
127      * Page requires transformation, since XSL is not welcome by the client?
128      * @param agent User agent of the client
129      * @param types Media types
130      * @return TRUE if the page requires transformation
131      */
132     private boolean xslAccepted(final UserAgent agent,
133         final TypesMatcher types) {
134         return agent.isXsltCapable()
135             && types.accepts(MediaType.APPLICATION_XML);
136     }
137 
138 }
